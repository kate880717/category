const DRIVE_FOLDER_NAME = '분류박물관_녹음파일'; // ============================================================ // doGet / doPost - CORS & JSONP 지원 // ============================================================ /** * GET 요청 처리 * - ?action=getAllData → 모든 시트 데이터 반환 * - JSONP callback 지원 (?callback=xxx) */ function doGet(e) { const params = e.parameter || {}; const action = params.action; let result; try { if (action === 'getAllData') { result = { success: true, data: getAllData() }; } else if (action === 'ping') { result = { success: true, msg: 'pong', time: new Date().toISOString() }; } else { result = { success: false, error: 'Unknown action: ' + action }; } } catch (err) { result = { success: false, error: err.toString() }; } // JSONP 지원 (CORS 우회용) if (params.callback) { return ContentService .createTextOutput(params.callback + '(' + JSON.stringify(result) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } /** * POST 요청 처리 - 결과 저장, 녹음 저장 * Apps Script는 자동으로 CORS 허용함 (웹앱이 모든 사용자 접근 시) */ function doPost(e) { let result; try { // POST body parsing const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === 'saveResult') { result = { success: true, ...saveResult(payload) }; } else if (action === 'saveRecording') { result = { success: true, ...saveRecording(payload) }; } else { result = { success: false, error: 'Unknown action: ' + action }; } } catch (err) { result = { success: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); return values.slice(1).filter(row => row[0] !== '' && row[0] !== null).map(row => { const obj = {}; headers.forEach((h, i) => { let v = row[i]; // boolean 값 정규화 if (typeof v === 'string') { if (v === 'TRUE' || v === 'true') v = true; else if (v === 'FALSE' || v === 'false') v = false; } obj[h] = v; }); return obj; }); } // ============================================================ // 학습 결과 저장 // ============================================================ function saveResult(payload) { const ss = SpreadsheetApp.getActiveSpreadsheet(); return {}; } // ============================================================ // 녹음 파일 저장 (Drive + Sheets) // ============================================================ function saveRecording(payload) { let driveUrl = ''; // Drive에 음성 파일 저장 if (payload.audioBase64 && payload.audioBase64.length > 0) { const folder = getOrCreateFolder(DRIVE_FOLDER_NAME); const decoded = Utilities.base64Decode(payload.audioBase64); const mimeType = payload.mimeType || 'audio/webm'; const ext = mimeType.split('/')[1].split(';')[0]; const safeName = (payload.studentName || 'unknown').replace(/[^\w가-힣]/g, ''); const fileName = `${payload.studentId || 'X'}_${safeName}_${payload.phase}_${payload.questionId}_${new Date().getTime()}.${ext}`; const blob = Utilities.newBlob(decoded, mimeType, fileName); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); driveUrl = file.getUrl(); } // Sheet에 텍스트 + URL 기록 const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES.RECORDINGS); sheet.appendRow([ new Date(), payload.studentId || '', payload.studentName || '', payload.phase === 'pre' ? '사전' : '사후', payload.questionId || '', payload.question || '', payload.transcript || '', driveUrl ]); return { driveUrl: driveUrl }; } function getOrCreateFolder(name) { const folders = DriveApp.getFoldersByName(name); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(name); } // ============================================================ // 메뉴 // ============================================================ function onOpen() { SpreadsheetApp.getUi() .createMenu('🔍 분류박물관') .addItem('① 시트 셋업', 'setupSheets') .addItem('② 웹앱 URL 보기', 'showWebAppUrl') .addItem('③ 연결 테스트', 'testConnection') .addSeparator() .addItem('학습결과 시트 비우기', 'clearResults') .addItem('녹음기록 시트 비우기', 'clearRecordings') .addToUi(); } function showWebAppUrl() { const url = ScriptApp.getService().getUrl(); if (!url) { SpreadsheetApp.getUi().alert( '⚠️ 아직 배포가 안 됐어요.\n\n' + '우측 상단 [배포] → [새 배포] → 톱니바퀴 → [웹 앱]\n' + '액세스: 모든 사용자\n실행: 나\n\n' + '배포 후 다시 눌러보세요.' ); } else { SpreadsheetApp.getUi().alert('🔗 웹앱 URL:\n\n' + url + '\n\n이 URL을 Claude에게 알려주세요.'); } } function testConnection() { const ui = SpreadsheetApp.getUi(); try { const data = getAllData(); const counts = Object.entries(data).map(([k, v]) => `• ${k}: ${v.length}개`).join('\n'); ui.alert('✅ 연결 정상!\n\n' + counts); } catch (err) { ui.alert('❌ 오류:\n\n' + err.toString()); } } function clearResults() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES.RESULTS); if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).clearContent(); } SpreadsheetApp.getUi().alert('학습결과 시트를 비웠어요.'); } function clearRecordings() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAMES.RECORDINGS); if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).clearContent(); } SpreadsheetApp.getUi().alert('녹음기록 시트를 비웠어요.'); }
